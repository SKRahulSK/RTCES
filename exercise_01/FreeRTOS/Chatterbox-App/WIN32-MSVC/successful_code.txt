  /* Standard includes. */
#include <stdio.h>


/* Kernel includes. */
#include "FreeRTOS.h"
#include "task.h"
#include "timers.h"

/* Priorities at which the tasks are created.
 */
#define TASK_CHATTERBOX_PRIORITY_TASK1 3
#define TASK_CHATTERBOX_PRIORITY_TASK2 2
#define TASK_CHATTERBOX_PRIORITY_TASK3 1

 /* Output frequencey, time between the output of sting on the console for each task
  */
#define TASK_CHATTERBOX_OUTPUT_FREQUENCY_MS 1000
  /*-----------------------------------------------------------*/

  /*
	*  Data structure to pass the required data to the task
	*  task_number - This is used to display the task number of eack task
	*  task_flag - This is used to distiguish  between the task which run infinitely (task1 and task2) and the task(task3) which stops after 5 executions.
	*  task_count - This is used internally to count the number of executions.
	*  task_handle - This is used internally to stop a particular task whenever required.
	*/
typedef struct {
	unsigned  task_number;
	unsigned  task_flag;
	unsigned  task_count;
	TaskHandle_t task_handle;
} taskParams_t;

/*
 * C function (prototype) for task
 */

void chatterboxTask(void* pvParameters);

/*-----------------------------------------------------------*/

/*** SEE THE COMMENTS AT THE TOP OF THIS FILE ***/
void main_exercise(void)
{
	/*
	 * Initialize data structures for all the three tasks
	 */
	taskParams_t task1_params = { 1, 0, 0, NULL }, task2_params = { 2,0, 0, NULL }, task3_params = { 3, 1, 0, NULL };

	/*
	 * Create the three task instances.
	 */
	xTaskCreate(chatterboxTask, "Chatterbox Task1", 1000, &task1_params, TASK_CHATTERBOX_PRIORITY_TASK1, task1_params.task_handle);
	xTaskCreate(chatterboxTask, "Chatterbox Task2", 1000, &task2_params, TASK_CHATTERBOX_PRIORITY_TASK2, task2_params.task_handle);
	xTaskCreate(chatterboxTask, "Chatterbox Task3", 1000, &task3_params, TASK_CHATTERBOX_PRIORITY_TASK3, task3_params.task_handle);

	/*
	 * Start the task instances.
	 */
	vTaskStartScheduler();
	/* If all is well, the scheduler will now be running, and the following
	line will never be reached.  If the following line does execute, then
	there was insufficient FreeRTOS heap memory available for the idle and/or
	timer tasks	to be created.  See the memory management section on the
	FreeRTOS web site for more details. */
	for (;; );
}
/*-----------------------------------------------------------*/

/*
 * C function for tasks
 */

void chatterboxTask(void* pvParameters)
{
	taskParams_t *task_params = (taskParams_t *)pvParameters;

	while (1)
	{
		/*
		 * Check if the task is Task number3 and increment the execution count
		 */
		if (task_params->task_flag == 1 && task_params->task_count <= 4)
		{
			task_params->task_count++;
			printf("TASK%u\n", task_params->task_number);
			if (task_params->task_count == 4)
			{
				/* Stop the task three, once the the execution count reches 5*/
				printf("TASK%u Stopping now\n", task_params->task_number);
				vTaskDelete(task_params->task_handle);
				task_params->task_handle = NULL;
			}
			
		}
		/* Run the task1 and task2 infinitely*/
		else if (task_params->task_flag == 0)
		{
			printf("TASK%u\n", task_params->task_number);
		}
		
		/*Sleep for user defined time*/
		vTaskDelay(pdMS_TO_TICKS(TASK_CHATTERBOX_OUTPUT_FREQUENCY_MS));
	};
}
